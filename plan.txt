FUNCTION dijkstra(start, goal)
    workingNodes = [start]
    selectedNodes = []
    while workingNodes.length !== 0
        workingNodes = insertionSort(workingNodes, workingNodes.length)
        current = workingNodes[0]
        if current == goal
            return generatePath(current)
        endif
        selectedNodes.push(current)
        neighbours = adjacentNodes(current)
        for each neighbour of neigbours
            if neighbour.weight + current.workingValue < neighbour.workingValue
                neighbour.workingValue = neighbour.weight + current.workingValue
                if !workingNodes.includes(neighbour)
                    workingNodes.push(neighbour)
                endif
            endif
        endloop
    endwhile
ENDFUNCTION
